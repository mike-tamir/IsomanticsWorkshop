    def normalize(matrix):
        """Normalize the rows of a matrix"""
        matrix_norm = np.linalg.norm(matrix, axis=1)
        matrix_normed = matrix / np.repeat(matrix_norm, matrix.shape[1]). \
            reshape(matrix.shape)
        return matrix_norm, matrix_normed
